/*   Constructues de proyectos  */

dotnet new sln                // Crea una solución nueva
dotnet new classlib           // Crea un proyecto de libreria estandar
dotnet new mstest             // Crea un proyecto de pruebas  MSTest unit test
dotnet new xunit              // Crea un proyecto de pruebas XUnit
dotnet new web                // Crea un proyecto web
dotnet new mvc                // Construye un nuevo proyecto mvc Razor pages
dotnet new mvc -o Presentacion
dotnet new angular            // Construye un nuenvo proyecto de angular
dotnet new react              // Construye un nuevo proyecto de react
dotnet new reactredux         // construye un nuevo proyecto de react con redux
dotnet new webapi             // construye un nuevo proyecto rest api
// Tipo de autenticación  None , IndividualB2C(azure), SingleOrg, Windows
dotnet new webapi -au|--auth <AUTHENTICATION_TYPE>
dotnet new console            // Crea un proyecto de consona nuevo

/*   Ejecución  */

dotnet restore                // Restaura las dependencias de un proyecto
dotnet run                    // Compila y ejecuta un proyecto


dotnet build                  // Construcción del proyecto
dotnet public                 // publicación del proyecto
dotnet test                   // Ejecuta las pruebas del proyecto

/*   Ejecutar en el package.json  */
"scripts": {
    // -c {Debug|Release} -p es el proyecto
    "start": "dotnet run -c Debug -p .\\src\\MyApplication.Web\\"
    // Ejecutar en modo obervación (actualiza los cambios)
    "rnw": "dotnet watch --project .\\blgApi\\blgApi.csproj run"
}


/*   Solución   */
dotnet new sln                              // Crea una nueva solución con el nombre de la carpeta que lo contiene
dotnet new sln -o <directory>               // Crea tanto el directorio como la solución con el mismo nombre
dotnet new sln --name <nombre>              // Crea una solución con el nombre indicado
dotnet sln add <path-to-project.csproj>     // Agrega el proyecto a la solución
                                            // Ejemplo dotnet sln add "src\CliApp\CliApp.csproj"
dotnet sln remove <path-to-project.csproj>  // Remueve el proyecto de la solución
dotnet sln list                             // listado de los proyectos de la solución

Ejemplos de la creación de un proyecto

// Crea la carpeta newproject y la solución newproject.sln
dotnet new sln -o newproject
// crea un proyecto mvc dentro de la carpeta presentacion
dotnet new mvc -o presentacion
// Crea un proyecto de clases dentro de la carpeta capadatos
dotnet new classlib -o capadatos
// Crea un proyecto de pruebas
dotnet new mstest -o test
// Agrega el proyecto presentación a la solución
/*lnx*/     dotnet sln newproject.sln add .\presentacion\presentacion.csproj
/*win*/     dotnet sln newproject.sln add  .\\presentacion\\presentacion.csproj
// Agrega el proyecto de clases a la solucion
/*lnx*/     dotnet sln newproject.sln add .\capadatos\capadatos.csproj
/*win*/     dotnet sln newproject.sln add .\\capadatos\\capadatos.csproj
// Agrega el proyecto de pruebas a la solución
/*lnx*/     dotnet sln newproject.sln add .\test\test.csproj
/*win*/     dotnet sln newproject.sln add .\\test\\test.csproj
/******* Agregar al proyecto de presentación una referencia del proyecto de clases******/
/*lnx*/     dotnet add .\presentacion\presentacion.csproj reference .\capadatos\capadatos.csproj
/*win*/     dotnet add .\\presentacion\\presentacion.csproj reference .\\capadatos\\capadatos.csproj
/******* Agregar al proyecto de pruebas una referencia del proyecto de clases******/
/*lnx*/     dotnet add .\test\test.csproj reference .\capadatos\capadatos.csproj
/*win*/     dotnet add .\\test\\test.csproj reference .\\capadatos\\capadatos.csproj

/* Remover un proyecto de la solución*/
dotnet sln todo.sln remove todo-app/todo-app.csproj

// Ejecuta las pruebas
dotnet test ~/projects/test1/test1.csproj
// Genera el resultado en un archivo en el formato trx
dotnet test ~/projects/test1/test1.csproj --logger:trx
// Ejecuta las pruebas y deja el archivo en la ruta indicadda
dotnet test ~/projects/test1/test1.csproj -l:trx;LogFileName=C:\temp\TestOutput.xml
// Pruebas con xUnit
dotnet xunit -xml output.xml

// Ejecuta el proyecto de inicio en modo watch, -p es ejecutar el proyecto
dotnet watch run -p ./presentacion/presentacion.csproj

// Publicacion
dotnet publish --output "<output-path>" --configuration release
dotnet publish .\MyWebApp.Client\MyWebApp.Client.csproj -o ..\build -c Release

Solución problemas de CORS

El Intercambio de Recursos de Origen Cruzado (CORS) s un mecanismo que utiliza cabeceras HTTP adicionales para permitir que un user agent obtenga permiso para acceder a recursos seleccionados desde un servidor, en un origen distinto (dominio) al que pertenece. Un agente crea una petición HTTP de origen cruzado cuando solicita un recurso desde un dominio distinto, un protocolo o un puerto diferente al del documento que lo generó.

Net Core

Configuración en Startup.cs

// Archivo : Startup.cs

public void ConfigureServices(IServiceCollection services){
            services.AddCors(options =>
            {
                options.AddPolicy("CorsPolicy",
                     builder => builder.AllowAnyOrigin()
                    .AllowAnyMethod()
                    .AllowAnyHeader()
                    .AllowCredentials());
            })
            .
            .
            .
}

public void Configure(IApplicationBuilder app, IHostingEnvironment env){
            if (env.IsDevelopment())
            {
                app.UseDeveloperExceptionPage();
            }
            else
            {
                app.UseHsts();
            }
            // Colocar la politica siempre antes de UseMvc()
            app.UseCors("CorsPolicy");
            app.UseHttpsRedirection();
            app.UseAuthentication();
            app.UseMvc();
}

Solución cliente

En esta solución usamos axios como cliente http.

// Crear un nuevo archivo con la instancia de axios
import axios from 'axios';
export default {
    function fncReques(baseUrl, token){
        return axios.create({
                baseURL: baseUrl,
                headers: {
                    'Authorization' : `Bearer ${token}`,
                    'Access-Control-Allow-Origin' : '*',
                    'Access-Control-Allow-Methods': '*',
                    'Content-Type': 'application/json',
                    'Access-Control-Allow-Headers': 'access-control-allow-headers,access-control-allow-methods,access-control-allow-origin,authorization'
                }
        });
    }
}

Configurar certificado de seguridad

Configuración https

En el archivo Program.cs se configuran las opciones del servicio web kestrel para que pueda leer un certificado autofirmado y se puedan realizar las peticiones utilizando HTTP/2

public class Program
{
    public static void Main(string[] args)
    {
        CreateWebHostBuilder(args).Build().Run();
    }

    public static IWebHostBuilder CreateWebHostBuilder(string[] args) =>
            WebHost.CreateDefaultBuilder(args)
            .UseKestrel(options =>{
                options.Listen(IPAddress.Any, 8765);
                options.Listen(IPAddress.Loopback, 7689, listenOptions =>{
                        listenOptions.Protocols = HttpProtocols.Http2;
                        listenOptions.UseHttps("certificado.pfx", "1qaz2wsx'pl,@123#$%ijnuhbygvABQER123@#");
                });
            })
            .UseStartup<Startup>();
}

En el ejemplo anterior se indica que redirija las peticiones HTTP por el puerto 8765 y HTTPS/2 por el puerto 7689, además utilice el archivo certificado.pfx que se encuentra en el directorio raiz del proyecto

Redireccionar todas las peticiones https

En el archivo Startup.cs se realiza el último ajuste para que todas las peticiones sean mediante https

public void Configure(IApplicationBuilder app, IHostingEnvironment env)
{
    if (env.IsDevelopment())
    {
        app.UseDeveloperExceptionPage();
    }
    else
    {
    }

    // Usar http/2
    app.UseHsts();
    // Redireccionar todas las peticiones a https
    app.UseHttpsRedirection();
    app.UseMvc();
}
