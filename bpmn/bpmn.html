Uso en la aplicación

import BpmnViewer from "bpmn-js";

var viewer = new BpmnViewer({
  container: "#canvas"
});

viewer.importXML(pizzaDiagram, function(err) {
  if (!err) {
    console.log("success!");
    viewer.get("canvas").zoom("fit-viewport");
  } else {
    console.log("something went wrong:", err);
  }
});
