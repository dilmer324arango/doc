Vue.component('vap-button', {   
    data : function(){
        return {
            count: 0
        }
    },
    props: ['title'],
    template : '<button v-on:click="count++">{{this.title}} {{ count }} </button>'
});

Plantilla Componente ES6

<template>
    <div>
        <p>Contenido del componenete {{strCampo}}</p>
        <p>Propiedad uno {{this.strProp}}</p>
    </div>
</template>    
<script>
    export default {
        name : vapComponente,
        data (){
            return {
                strCampo : ''
            }
        },
        props {
            strProp {
                type        : String,
                required    : true,                
            },
            nmbProp {
                type        : Number,
                required    : false, 
                default     : 0
            },
            blnProp {
                type        : Boolean,
                required    : true, 
            },
            aryProp {
                type        : Array,
                required    : false,
                default     : function () {  // Object or array defaults must be returned from
                    return [];               // a factory function 
                }
            },
            objProp {
                type        : Object,
                required    : true, 
                default     : function () {  // Object or array defaults must be returned from
                    return {};               // a factory function 
                }
            }
        },
        methods : {
            fncMethodUno : function (){
                return this.vaUno;
            }
        },
        computed: {
            fncComputed : function(){
                return this.varUno;
            }
        },
        watch: {  
            fncWatch: function () {
                return this.varUno;
            }
        },
    }
</script>
