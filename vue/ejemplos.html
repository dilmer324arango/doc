Eventos formulario

Este ejemplo permite controlar los eventos de teclado para un formulario.

<!DOCTYPE html>
<html lang="es">
<head>
    <meta charset="UTF-8">
    <title>Vue | Computed properties </title>
    <link rel="stylesheet" href="../../css/bootstrap.min.css">
</head>
<body>
    <div class="container" id="main">
        <div >
            <form  id="form_1" action="" @submit.prevent="fncEnviarDataos()" autocomplete="off">
                <div class="form-group">
                    <input 
                    id="inp_1"
                    name ="inp_1" 
                    v-model="elements.inp_1"
                    autocomplete="off"                    
                    v-on:keydown.prevent.esc.up.down.enter.tab = "fnSetKeyPress($event)"                    
                    type="text"                     
                    required="required"/>
                    <label for="inp_1" class="control-label">Nombre<span>*</span></label><i class="bar"></i>
                </div>
                <div class="form-group">
                    <input 
                    id="inp_2"
                    name="inp_2"
                    v-model="elements.inp_2"
                    autocomplete="off"
                    v-on:keydown.prevent.esc.up.down.enter.tab = "fnSetKeyPress($event)"
                    type="password" 
                    required="required"/>
                    <label for="inp_2" class="control-label">password<span>*</span></label><i class="bar"></i>
                </div>
                <div class="form-group">
                    <input 
                    id="inp_3"
                    name ="inp_3"
                    v-model="elements.inp_3"
                    maxlength="5"
                    autocomplete="off"
                    v-on:keydown= "fncKeyPressNumber($event)"                   
                    type="text" 
                    required="required"/>
                    <label for="inp_3" class="control-label">Edad<span>*</span></label><i class="bar"></i>
                </div>
                
                <div class="form-group">
                    <input 
                    id="inp_4"
                    name ="inp_4"
                    v-model="elements.inp_4"
                    autocomplete="off"
                    v-on:keydown= "fncKeyPressDecimal($event)"
                    type="text" 
                    required="required"/>
                    <label for="inp_4" class="control-label">Salario<span>*</span></label><i class="bar"></i>
                </div>
                <div class="form-group">
                    <input 
                    id="inp_5"
                    name="inp_5"
                    v-model="elements.inp_5"
                    autocomplete="off"
                    v-on:keydown.prevent.esc.up.down.tab = "fnSetKeyPress($event)"
                    v-on:keydown.prevent.enter="fncEnviarDataos"                    
                    type="text" 
                    required="required"/>
                    <label for="inp_5" class="control-label">e-mail<span>*</span></label><i class="bar"></i>
                </div>
                <div class="button-container" v-if="fncTodaLaInformacion">
                    <button   type="submit" class="button">
                        <span>Enviar Datos</span>
                    </button>
                </div>    

            </form>
        </div>
        <pre>
            {{$data | json }}
        </pre>
    </div>
    <script src="./lib/vue.min.js"></script>
    <script type="text/javascript" src="app.js"></script>
</body>
</html>

Archivo app.js

// En el separador puede colocar el tag y la clase  div.form-group
// con lo anterir excluye los botones 
var vm = new Vue({
    el: '#main'
    , data: {
          last                  : 0
        , formName              : "form_1"
        , firstInputType        : "input"
        , tagSeparator          : "div.form-group"
        , elements              : {inp_1:"", inp_2:"", inp_3:"", inp_4:"", inp_5:""}
        , o                     : {bac: 8, tab:9, ent:13, esc:27, up:38, down:40, dotpat:110, dot:190}
        , strLastInputChange    : ""
    }
    , methods: {
        fncEnviarDataos: function () {
            if ( this.fncValidarCampos() ) {
                alert('Enviar datos')
            } else {
                alert('Todos los campos con * son requeridos ')
            }
        }
        , fnSetKeyPress: function (e) {
            var name  = e.target.id.split('_')[0] + '_';
            var nmbId = parseInt(e.target.id.split('_')[1]); 
            var nmbUp   = nmbId - 1; 
            var nmbDown = nmbId + 1; 
            var obj = {};
            switch(e.keyCode){
               case this.o.esc: 
                    if(this.elements[e.target.id]){this.elements[e.target.id] = '';}
                    obj = document.getElementById(e.target.id);
                    if(obj){
                        obj.value = '';
                        obj.focus();
                    }
                break; 
                case this.o.up :
                    obj = document.getElementById(name + nmbUp);
                    if(obj){
                        obj.focus();
                    }else{
                       obj = document.getElementById( name + this.last );
                       obj.focus();
                    }
                break; 
                case this.o.down : case this.o.ent : case this.o.tab :
                    obj = document.getElementById(name + nmbDown);
                    if (obj) {
                        obj.focus();
                    } else {
                        this.fncDefaultFocus();
                    }
                break;                 
            }            
        }
        , fncKeyPressNumber: function(e){
            var k =  e.keyCode;
            var obj = document.getElementById(e.target.id);
            if(  (k === this.o.ent ) || (k === this.o.esc ) || (k === this.o.up ) || (k === this.o.down) ){
                if(this.strLastInputChange === e.target.id){Vue.nextTick(function () { vm.fncValidarNumber(obj);  });}
                this.fnSetKeyPress(e);
                return true;
            }else if( (k === this.o.tab ) ){
                e.preventDefault();                
                if(this.strLastInputChange === e.target.id){Vue.nextTick(function () { vm.fncValidarNumber(obj);  });}
                this.fnSetKeyPress(e);
                return true;
            }else if(  (k > 31 && (k < 48 || k > 57) && (k < 96 || k > 105) )  ) {  
                e.preventDefault();
            }else{
                this.strLastInputChange = e.target.id; 
                return true;
            }            
        }
        , fncKeyPressDecimal : function(e){
            var k =  e.keyCode;
            var obj = document.getElementById(e.target.id);
            if(  (k === this.o.ent ) || (k === this.o.esc ) || (k === this.o.up ) || (k === this.o.down) ){
                if(this.strLastInputChange === e.target.id ) {Vue.nextTick(function () { vm.fncValidarDecimal(obj);  });}
                this.fnSetKeyPress(e);
                return true;
            }else if( (k === this.o.tab ) ){
                e.preventDefault();
                if(this.strLastInputChange === e.target.id ) {Vue.nextTick(function () { vm.fncValidarDecimal(obj);  });}
                this.fnSetKeyPress(e);
                return true;
            }else if(  (k > 31 && (k < 48 || k > 57) && (k < 96 || k > 105) ) && (k < this.o.dotpat || k >this.o.dotpat ) && (k < this.o.dot || k >this.o.dot) ) {  
                e.preventDefault();
            }else{
                
                var strValue = obj.value;
                if(strValue.includes(".")){
                    let str = strValue.split(".");  
                    if(str.length >2){
                        if(str[1].length>4){
                            obj.value = str[0] + "." + str[1].slice(0,3);
                        }else{
                            obj.value = str[0] + "." + str[1];
                        }                        
                    } 
                    // this.fncValidarDecimal(obj);                     
                } 
                this.strLastInputChange = e.target.id;               
                return true;
            }   
        }
        , fncValidarNumber : function(obj){
            var strValue = obj.value;
            if(strValue.length !== 0){
                obj.value = parseInt(strValue);
                this.elements[obj.id] = parseInt(strValue);                
            }
            this.strLastInputChange ="";
        }
        , fncValidarDecimal   : function(obj){
            var strValue = obj.value;
            if(strValue.includes(".")){
                let str = strValue.split("."); 
                if(str[0].length == 0){str[0] = "0"}
                if(str[1].length < 4){
                    str[1] = str[1] + "00000";
                }
                let value = str[0] + "." + str[1].slice(0,3);
                obj.value = parseFloat(value);
                this.elements[obj.id] = parseFloat(value);                
            }
            this.strLastInputChange ="";
            return true;
        }        
        , fncCleanAll: function () {
            for(let i = 0 ; i< this.last; i++){             
                if(this.elements[i]){ this.elements[i] = '';}  
            }
            this.fncDefaultFocus(); 
        }
        , fncDefaultFocus: function () {
            var inpNombre = document.getElementById(this.formName).querySelectorAll(this.firstInputType)[0];
            // var inpNombre = this.$els.inpFocus;
            Vue.nextTick(function () {
                inpNombre.focus();
            });
        }
        , fncValidarCampos : function(){
            for (var key in this.elements) {
                if(!this.elements[key]){return false}                    
            }
            return true;
        }
    }
    , computed: {
        // Se tiene acceso a las propiedades del modelo, son accesibles desde la vista y reactivas 
        fncTodaLaInformacion: function () {
            return this.fncValidarCampos();
        }
        
    }
    , mounted: function () {
        this.fncDefaultFocus();     
        this.last = document.getElementById(this.formName).querySelectorAll(this.tagSeparator).length;
    }  
})
